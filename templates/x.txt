        function canMakeRecipe(recipe) {
            const available = getAvailableIngredients();
            return recipe.ingredients.every(ingredient => {
                const ingredientLower = ingredient.toLowerCase();
                const ingredientSingular = ingredientLower.endsWith('s') ? ingredientLower.slice(0, -1) : ingredientLower;
                const ingredientPlural = !ingredientLower.endsWith('s') ? ingredientLower + 's' : ingredientLower;
                
                return available.some(item => 
                    item === ingredientLower || 
                    item === ingredientSingular || 
                    item === ingredientPlural
                );
            });
        }

        function filterRecipes() {
            const searchTerm = document.querySelector('.search-input').value.toLowerCase();
            const activeCategory = Array.from(document.querySelectorAll('.filter-btn')).find(btn => btn.classList.contains('active'))?.textContent || 'All';
            filterByCategory(activeCategory);
        }

        function filterByCategory(category) {
            // Update active button state
            const buttons = document.querySelectorAll('.filter-btn');
            buttons.forEach(btn => {
                btn.classList.remove('active');
                if (btn.textContent === category || 
                    (category.toLowerCase() === 'all' && btn.textContent === 'All Recipes') ||
                    (category === 'available' && btn.textContent === 'Using Available Ingredients') ||
                    (category === 'saved' && btn.textContent === 'Saved Recipes')) {
                    btn.classList.add('active');
                }
            });

            // Filter recipes
            const searchTerm = document.querySelector('.search-input').value.toLowerCase();
            let filteredRecipes = [...recipes];

            // Apply category filter
            if (category.toLowerCase() !== 'all') {
                if (category === 'available') {
                    filteredRecipes = filteredRecipes.filter(recipe => canMakeRecipe(recipe));
                } else if (category === 'saved') {
                    filteredRecipes = filteredRecipes.filter(recipe => savedRecipes.has(recipe.id));
                } else {
                    filteredRecipes = filteredRecipes.filter(recipe => recipe.category === category);
                }
            }

            // Apply search filter if exists
            if (searchTerm) {
                filteredRecipes = filteredRecipes.filter(recipe =>
                    recipe.title.toLowerCase().includes(searchTerm) ||
                    recipe.ingredients.some(ing => ing.toLowerCase().includes(searchTerm))
                );
            }

            // Update display
            displayFilteredRecipes(filteredRecipes);
        }

                                    const isAvailable = getAvailableIngredients().includes(ing.toLowerCase());
                                    return `
                                        <div class="ingredient-item">
                                            <i class="fas ${isAvailable ? 'fa-check-circle text-success' : 'fa-times-circle text-danger'}"></i>
                                            <span>${ing}</span>
                                        </div>
                                    `;
                                }).join('')}
                            </div>
                            <div class="recipe-actions">
                                <button class="btn-recipe btn-view" onclick="viewRecipe(${recipe.id})">
                                    <i class="fas fa-eye"></i> View Recipe
                                </button>

                                </button>

                `;

        }

        function viewRecipe(recipeId) {
            const recipe = recipes.find(r => r.id === recipeId);
            if (!recipe) return;

            const modalContent = document.getElementById('recipeModalContent');
            modalContent.innerHTML = `
                <div class="recipe-detail">
                    <img src="${recipe.image}" alt="${recipe.title}" style="width: 100%; height: 300px; object-fit: cover; border-radius: 10px;">
                    <h3 class="mt-4">${recipe.title}</h3>
                    <div class="recipe-info mb-4">
                        <span><i class="fas fa-clock"></i> ${recipe.time}</span>
                        <span><i class="fas fa-utensils"></i> ${recipe.difficulty}</span>
                        <span><i class="fas fa-users"></i> ${recipe.servings} servings</span>
                    </div>
                    <h4>Ingredients:</h4>
                    <ul class="list-unstyled">
                        ${recipe.ingredients.map(ing => {
                            const ingLower = ing.toLowerCase();
                            const ingSingular = ingLower.endsWith('s') ? ingLower.slice(0, -1) : ingLower;
                            const ingPlural = !ingLower.endsWith('s') ? ingLower + 's' : ingLower;
                            const isAvailable = getAvailableIngredients().some(item => 
                                item === ingLower || 
                                item === ingSingular || 
                                item === ingPlural
                            );
                            return `
                                <li class="mb-2">
                                    <i class="fas ${isAvailable ? 'fa-check-circle text-success' : 'fa-times-circle text-danger'}"></i>
                                    ${ing}
                                </li>
                            `;
                        }).join('')}
                    </ul>
                    <h4>Instructions:</h4>
                    <ol>
                        ${recipe.instructions.map(step => `<li class="mb-2">${step}</li>`).join('')}
                    </ol>
                </div>
            `;

            const modal = new bootstrap.Modal(document.getElementById('recipeModal'));
            modal.show();
        }

        function checkAvailableIngredients(recipe) {
            const availableIngredients = new Set();
            const inventory = JSON.parse(localStorage.getItem('inventoryItems') || '[]');
            
            // Helper function to get singular form of a word
            function getSingular(word) {
                if (word.endsWith('s')) {
                    return word.slice(0, -1);
                }
                return word;
            }

            // Helper function to get plural form of a word
            function getPlural(word) {
                if (!word.endsWith('s')) {
                    return word + 's';
                }
                return word;
            }

            // Check each ingredient against inventory
            recipe.ingredients.forEach(ingredient => {
                const ingredientLower = ingredient.toLowerCase();
                const ingredientSingular = getSingular(ingredientLower);
                const ingredientPlural = getPlural(ingredientLower);

                // Check if any inventory item matches the ingredient (singular or plural)
                const hasIngredient = inventory.some(item => {
                    const itemName = item.fruit.toLowerCase();
                    return itemName === ingredientLower || 
                           itemName === ingredientSingular || 
                           itemName === ingredientPlural;
                });

                if (hasIngredient) {
                    availableIngredients.add(ingredient);
                }
            });

            return availableIngredients;
        }
